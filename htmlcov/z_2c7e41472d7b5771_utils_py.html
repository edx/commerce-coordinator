108 ''' 

109 Checks that uut produces expected_request and expected_output given input_kwargs and mock_response. 

110 

111 Mocks any calls by requests to self.mock_url. Returns mock_response for those calls as JSON. 

112 

113 Optionally, checks headers match self.expected_headers. 

114 

115 Args: 

116 uut (callable): Unit under test. Calls an external API using the `requests` library. 

117 input_kwargs (dict): kwargs to provide uut. 

118 expected_request (dict): Expected request of uut to external API given input_kwargs. POST requests will be 

119 converted to JSON. 

120 request_type (str): 'query_string' if the expected_request is of type application/x-www-form-urlencoded; 

121 'json' for type 'application/json'. Defaults to 'json'. 

122 expected_headers (dict): Expected headers of uut to external API. 

123 mock_url (str): URL of external API to mock. 

124 mock_response (dict): Mock response external API should provide uut given expected_request. Will be 

125 converted to JSON. 

126 mock_status (int): HTTP Status Code 

127 mock_method (str): String of the Mocked Method (GET, POST, etc) 

128 expected_output (object): Expected return value of uut given mock_response. 

129 

130 ''' 

131 

132 is_get = mock_method == 'GET' 

133 

134 # Use matcher for query params for GET requests: 

135 if is_get: 

136 expected_match = [ 

137 responses.matchers.query_param_matcher(expected_request) 

138 ] 

139 else: 

140 expected_match = [] 

141 

142 # Prepare external API's mock response: 

143 if not mock_response: 

144 mock_response = {} 

145 responses.add( 

146 method=mock_method, 

147 url=mock_url, 

148 status=mock_status, 

149 json=mock_response, 

150 match=expected_match, 

151 ) 

152 

153 # Get client output & request it built for external API. 

154 # Throw exceptions after asserts. 

155 exception_thrown = None 

156 try: 

157 output = uut(**input_kwargs) 

158 except Exception as exc: # pylint: disable=broad-exception-caught 

159 print_exc() 

160 # Will rethrow this exception later. 

161 exception_thrown = exc 

162 

163 # self.assertGreaterEqual(len(responses.calls), 1, "no responses were generated for this API") 

164 request = responses.calls[-1].request if responses.calls else None 

165 

166 # Perform checks: 

167 if expected_headers: 

168 self.assertGreaterEqual( 

169 request.headers.items(), 

170 expected_headers.items(), 

171 'Check external API called with expected headers' 

172 ) 

173 if expected_request: 

174 if is_get: 

175 request_dict = request.params 

176 elif request_type == 'query_string': 

177 request_dict = parse_qs(request.body, strict_parsing=True)

180 else: 

181 raise ValueError('request_type must be "query_string" or "json" ' + 

182 'for non-GET expected_request')

184 request_dict, 

185 expected_request, 

186 'Check external API called with expected input' 

187 ) 

188 if exception_thrown: 

189 raise exception_thrown 

190 # We can't check expected_output if exception_thrown. 

191 if expected_output: 

192 self.assertEqual( 

193 output, 

194 expected_output, 

195 'Check client returns expected output' 

196 )

282def random_unicode_str(ln: int, limit_unicode=True, weight_divisor=2): 

283 """ Generate a string of X characters guaranteed to include at least one non ASCII one. """ 

284 

285 uchars = ['\xe9', '\xf1', '\xfc', '\u0110', '\u0159', '\u016f', '\xc5', '\xdf', '\xe7', '\u0131', 

286 '\u0130', '\uff21', '\ufb04', '\u211a', '\xbd', '\u20ac', '\u20b9', '\xa5', '\u0416', '\u03b7', 

287 '\uae00', '\u0913', '\u0b15', '\u3058', '\u5b57', '\U0001f40d', '\U0001f496', '\u2652', '\u2658'] 

288 

289 # The above represents, Escapes used incase this file is ever re-encoded by accident. 

290 # ['é', 'ñ', 'ü', 'Đ', 'ř', 'ů', 'Å', 'ß', 'ç', 'ı', 'İ', 'Ａ', 'ﬄ', 'ℚ', '½', '€', '₹', '¥', 'Ж', 'η', '글', 

291 # 'ओ', 'କ', 'じ', '字', '🐍', '💖', '♒', '♘'] 

292 

293 chars = uchars + list(string.printable) 

294 

295 def _unicode_limiter(c: (int, str)): 

296 """ Map replacement function to limit unicode based on a divisor """ 

297 

298 if limit_unicode: 

299 if len(bytes(chars[c[0]], encoding='utf8')) > 1: 

300 return 1 

301 else: 

302 return abs(len(chars) / weight_divisor) 

303 else: 

304 return 1 

305 

306 weights = list(map(_unicode_limiter, enumerate(chars))) 

307 

308 retval = ''.join(random.choices(chars, weights=weights, k=ln - 1)) 

309 

310 retval += random.choice(uchars) # ensure we get one unicode no matter what. 

311 

312 assert len(retval) == ln 

313 return retval
