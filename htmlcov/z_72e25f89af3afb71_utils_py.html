107def extract_ct_order_information_for_braze_canvas(customer: Customer, order: Order): 

108 """ 

109 Utility to extract generic order information for braze canvas properties 

110 """ 

111 order_placed_on = order.last_modified_at 

112 formatted_order_placement_date = order_placed_on.strftime('%b %d, %Y') 

113 formatted_order_placement_time = order_placed_on.strftime("%I:%M %p (%Z)") 

114 # calculate subtotal by adding discount back if any discount is applied. 

115 # TODO: Post R0.1 add support for all discount types here. 

116 subtotal = (((order.total_price.cent_amount + 

117 order.discount_on_total_price.discounted_amount.cent_amount)) 

118 if order.discount_on_total_price else order.total_price.cent_amount) 

119 canvas_entry_properties = { 

120 "first_name": customer.first_name, 

121 "last_name": customer.last_name, 

122 "redirect_url": settings.LMS_DASHBOARD_URL, 

123 "view_receipt_cta_url": f"{settings.COMMERCE_COORDINATOR_URL}{reverse('frontend_app_ecommerce:order_receipt')}" 

124 f"?order_number={order.order_number}", 

125 "purchase_date": formatted_order_placement_date, 

126 "purchase_time": formatted_order_placement_time, 

127 "subtotal": format_amount_for_braze_canvas(subtotal), 

128 "total": format_amount_for_braze_canvas(order.total_price.cent_amount), 

129 } 

130 # TODO: Post R0.1 add support for all discount types here. 

131 if order.discount_codes and order.discount_on_total_price: 

132 canvas_entry_properties.update({ 

133 "discount_code": order.discount_codes[0].discount_code.obj.code, 

134 "discount_value": format_amount_for_braze_canvas( 

135 order.discount_on_total_price.discounted_amount.cent_amount), 

136 }) 

137 return canvas_entry_properties

150def has_full_refund_transaction(payment: Payment): 

151 """ 

152 Utility to determine is CT payment has an existing 'refund' transaction for the full 

153 charge amount 

154 """ 

155 # get charge transaction and get amount then check against refund. 

156 charge_amount = 0 

157 for transaction in payment.transactions: 

158 if transaction.type == TransactionType.CHARGE: 

159 charge_amount = transaction.amount 

160 if transaction.type == TransactionType.REFUND and transaction.amount == charge_amount: # pragma no cover 

161 return True 

162 return False

188def _create_retired_hash_withsalt(value_to_retire, salt): 

189 """ 

190 Returns a retired value given a value to retire and a hash. 

191 

192 Arguments: 

193 value_to_retire (str): Value to be retired. 

194 salt (str): Salt string used to modify the retired value before hashing. 

195 """ 

196 return hashlib.sha256( 

197 salt.encode() + value_to_retire.encode('utf-8') 

198 ).hexdigest()

201def create_retired_fields(field_value, salt_list, retired_user_field_fmt=RETIRED_USER_FIELD_DEFAULT_FMT): 

202 """ 

203 Returns a retired field value based on the original lowercased field value and 

204 all the historical salts, from oldest to current. The current salt is 

205 assumed to be the last salt in the list. 

206 

207 Raises :class:`~ValueError` if the salt isn't a list of salts. 

208 

209 Arguments: 

210 field_value (str): The value of the field to be retired. 

211 salt_list (list/tuple): List of all historical salts. 

212 

213 Yields: 

214 Returns a retired value based on the original field value 

215 and all the historical salts, including the current salt. 

216 """ 

217 if not isinstance(salt_list, (list, tuple)): 

218 raise SALT_LIST_EXCEPTION 

219 

220 return retired_user_field_fmt.format(_create_retired_hash_withsalt(field_value.lower(), salt_list[-1]))
